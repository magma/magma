# Palette

To use the Symphony palette, import the theme and access the palette with `symphony.palette`.

```js
import symphony from '../../../ui/theme/symphony';
```

export const ColorBlocks = props => {
  const {colors, title} = props;
  const classes = useStyles();
  return (
    <div className={classes.blocksRoot}>
      <Text className={classes.title} weight="medium">
        {title}
      </Text>
      <div className={classes.colors}>
        {colors.map(color => (
          <ColorBlock
            key={color.color}
            className={classes.colorBlock}
            color={color.color}
            name={color.name}
            code={color.code}
          />
        ))}
      </div>
    </div>
  );
};
