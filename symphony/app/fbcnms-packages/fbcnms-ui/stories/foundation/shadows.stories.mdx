# Elevation

To add elevation, import the theme and access the shadows with `symphony.shadows`.

```js
import symphony from '@fbcnms/ui/theme/symphony';
```

export const ShadowBlock = props => {
  const {shadow} = props;
  const classes = useStyles();
  return (
    <div className={classes.blockRoot}>
      <div
        className={classNames(
          classes.shadowBlock,
          classes[shadow.toLowerCase()],
        )}
      />
      <Text weight="medium">{shadow.toUpperCase()}</Text>
    </div>
  );
};

export const ShadowBlocksRoot = () => {
  const classes = useStyles();
  return (
    <div className={classes.root}>
      {Object.keys(symphony.shadows).map(shadow => (
        <ShadowBlock key={shadow} shadow={shadow} />
      ))}
    </div>
  );
};
